4. CHECK
Syntax
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
Insert
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
Error Example
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
INSERT INTO Student VALUES(2,15);
Error
Check constraint violated
Solution
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
INSERT INTO Student VALUES(2,18);

6. FOREIGN KEY
Parent Table

    deptid INT PRIMARY KEY,
    deptname VARCHAR(30)
);
Child Table
CREATE TABLE Student(
    id INT PRIMARY KEY,
    name VARCHAR(30),
    deptid INT,
    FOREIGN KEY(deptid)
    REFERENCES Department(deptid)
);

-- Create a new database
CREATE DATABASE ShopDB;

-- Switch to the new database
USE ShopDB;

-- Customers table
CREATE TABLE Customers (
    CustomerID INT PRIMARY KEY,
    Name VARCHAR(50),
    Email VARCHAR(50)
);

-- Orders table
CREATE TABLE Orders (
    OrderID INT PRIMARY KEY,
    CustomerID INT,
    Product VARCHAR(50),
    Quantity INT,
    FOREIGN KEY (CustomerID) REFERENCES Customers(CustomerID)
);

-- --------------------
-- INSERT OPERATIONS
-- --------------------

-- Correct insert
INSERT INTO Customers (CustomerID, Name, Email)
VALUES (1, 'Lahari', 'lahari@email.com');

-- Insert with missing Email
-- This works because Email allows NULL
INSERT INTO Customers (CustomerID, Name)
VALUES (2, 'Ravi');

-- Correct order insert
INSERT INTO Orders (OrderID, CustomerID, Product, Quantity)
VALUES (101, 1, 'Laptop', 5);

-- Foreign key violation
-- This will produce an error because CustomerID 99 doesn't exist
INSERT INTO Orders (OrderID, CustomerID, Product, Quantity)
VALUES (102, 99, 'Phone', 2);

-- --------------------
-- CRUD OPERATIONS
-- --------------------

-- CREATE
INSERT INTO Customers (CustomerID, Name, Email)
VALUES (3, 'Anjali', 'anjali@email.com');

-- READ
SELECT * FROM Customers;

SELECT * FROM Orders
WHERE CustomerID = 1;

-- UPDATE
UPDATE Customers
SET Email = 'lahari_new@email.com'
WHERE CustomerID = 1;

-- DELETE
DELETE FROM Customers
WHERE CustomerID = 3;

-- --------------------
-- DELETE, TRUNCATE, DROP
-- --------------------

-- DELETE: Removes selected rows
DELETE FROM Orders
WHERE CustomerID = 1;

-- Insert again before TRUNCATE
INSERT INTO Orders (OrderID, CustomerID, Product, Quantity)
VALUES (103, 1, 'Mouse', 2);

-- TRUNCATE: Removes all rows
TRUNCATE TABLE Orders;

-- DROP: Removes the table completely
DROP TABLE Orders;

-- --------------------
-- LIMIT and OFFSET
-- --------------------

-- Show first 2 customers
SELECT * FROM Customers
LIMIT 2;

-- Skip first 2 customers and display next 2
SELECT * FROM Customers
LIMIT 2 OFFSET 2;
